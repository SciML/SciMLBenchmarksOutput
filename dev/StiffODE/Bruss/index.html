function fast_bruss(N)

  u0 = zeros(N, N, 2)

    x = xyd_brusselator[I[1]]

    u0[I,1] = 22*(y*(1-y))^(3/2)
    u0[I,2] = 27*(x*(1-x))^(3/2)
  end
  return ODEProblem(f,u0,(0.,11.5),p)
end
